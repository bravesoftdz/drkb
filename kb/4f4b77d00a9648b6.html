Как очистить буфер клавиатуры?

01.01.2010
procedure EmptyKeyQueue;
var
  msg: TMsg;
begin
  while PeekMessage(msg, 0, WM_KEYFIRST, WM_KEYLAST, PM_REMOVE or PM_NOYIELD) do
    ;
end;
 
begin
  EmptyKeyQueue;
end.
