function GetDisplayDevice: string;
var
  lpDisplayDevice: TDisplayDevice;
begin
  lpDisplayDevice.cb := sizeof(lpDisplayDevice);
  EnumDisplayDevices(nil, 0, lpDisplayDevice , 0);
  Result:=lpDisplayDevice.DeviceString;
end;
