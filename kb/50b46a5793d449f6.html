Проверить, используется ли формат времени в 24 часа

01.01.2010
function Is24HourTimeFormat: Boolean;
 var
   DefaultLCID: LCID;
 begin
   DefaultLCID := GetThreadLocale;
   Result := 0 <> StrToIntDef(GetLocaleStr(DefaultLCID,
     LOCALE_ITIME,'0'), 0);
 end;
