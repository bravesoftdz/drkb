function DateExists(Date: string; Separator: char): Boolean;
var
  OldDateSeparator: Char;
begin
  Result := True;
  OldDateSeparator := DateSeparator;
  DateSeparator := Separator;
  try
    try
      StrToDate(Date);
    except
      Result := False;
    end;
  finally
    DateSeparator := OldDateSeparator;
  end;
end;

function ValidDate(const S: String): Boolean;
BEGIN
  Result := True;
  try
    StrToDate(S);
  except
    ON EConvertError DO
      Result := False;
  end;
END
