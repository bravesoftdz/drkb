function PWideToString(pw : PWideChar) : string;
var
  p : PChar;
  iLen : integer;
begin
   iLen := lstrlenw( pw ) + 1;
   GetMem( p, iLen );
 
   WideCharToMultiByte( CP_ACP, 0, pw, iLen, p, iLen * 2, nil, nil );
 
   Result := p;
   FreeMem( p, iLen );
end;
