function CheckSSE2: Boolean;
var
  TempCheck: dword;
begin
  TempCheck := 1;
  asm
  push ebx
  mov eax,1
  db $0F,$A2
  test edx,$4000000
  jz @NOSSE2
  mov edx,0
  mov TempCheck,edx
  @NOSSE2:
  pop ebx
  end;
  CheckSSE2 := (TempCheck = 0);
end;
