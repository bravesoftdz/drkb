function CheckSSE: Boolean;
var
  TempCheck: dword;
begin
  TempCheck := 1;
  asm
  push ebx
  mov eax,1
  db $0F,$A2
  test edx,$2000000
  jz @NOSSE
  mov edx,0
  mov TempCheck,edx
  @NOSSE:
  pop ebx
  end;
  CheckSSE := (TempCheck = 0);
end;
